192  template<typename IT,typename CTYPE>
193  void strip_reader::_read_interlace(size_t channel,
194  std::ifstream &stream,CTYPE &data) const
195  {
196  typedef typename CTYPE::value_type value_type;
197  //compute the size of a pixel in bytes
198  size_t pixel_size = std::accumulate(_bits_per_channel.begin(),
199  _bits_per_channel.end(),0)/8;
200 
201  //compute the offset from the begining of a pixel to the requested
202  //sample. The offset is given in bytes
203  size_t sample_offset = std::accumulate(_bits_per_channel.begin(),
204  _bits_per_channel.begin()+channel,
205  0)/8;
206 
207  //size of the sample type
208  size_t sample_size = sizeof(IT);
209 
210  IT sample_buffer;
211 
212  typename CTYPE::iterator piter = data.begin(); //pixel iterator
213  while(piter != data.end())
214  {
215  //loop over all strips
216  for(size_t strip=0;strip<_offsets.size();strip++)
217  {
218  size_t npixels = _byte_cnts[strip]/pixel_size;
219 
220  //set the stream to the offset of the actual strip
221  stream.seekg(_offsets[strip]+sample_offset,std::ios::beg);
222 
223  //loop over all pixels in the strip
224  for(size_t i=0;i<npixels; ++i)
225  {
226  stream.read(reinterpret_cast<char*>(&sample_buffer),sample_size);
227  stream.seekg(pixel_size-sample_size,std::ios::cur);
228  *piter++ = value_type(sample_buffer);
229  }
230  }
231  }
232  }
