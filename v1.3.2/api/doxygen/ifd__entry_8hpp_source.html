209  template<typename T> std::vector<T> ifd_entry::value(std::ifstream &stream)
210  {
211  using namespace pni::core;
212 
213  //create a vector of appropriate length
214  std::vector<T> result(this->size());
215  //save the original stream position
216  std::streampos orig_stream_pos = stream.tellg();
217 
218  //set the stream to the begining of the data section
219  stream.seekg(this->_data,std::ios::beg);
220 
221  //here comes the tricky part. Though the user defines the type T he
222  //wants to have the IFD entry not all entry types can be converted to T
223  //without loss of information. Thus wee need a selection function that
224  //picks for each T all acceptable types and throws an exception if the
225  //entry is of an incompatible type.
226  //
227  //This operation is carried out by the next function which cann choose
228  //the proper version by argument type deduction.
229  try{
230  this->_read_entry_data(result,stream);
231  }
232  catch(type_error &e)
233  {
234  e.append(EXCEPTION_RECORD);
235  //reset the stream to its original position
236  stream.seekg(orig_stream_pos,std::ios::beg);
237  throw e;
238  }
239 
240 
241  //reset stream to its original position
242  stream.seekg(orig_stream_pos,std::ios::beg);
243 
244  return result;
245  }

248  template<typename T> void ifd_entry::
249  _read_entry_data(std::vector<T> &r,std::ifstream &stream)
250  {
251  using namespace pni::core;
252 
253  if(this->_tid == ifd_entry_type_id::BYTE)
254  ifd_entry_reader<T,uint8>::read(r,stream);
255  else if(this->_tid == ifd_entry_type_id::SHORT)
256  ifd_entry_reader<T,uint16>::read(r,stream);
257  else if(this->_tid == ifd_entry_type_id::LONG)
258  ifd_entry_reader<T,uint32>::read(r,stream);
259  else if(this->_tid == ifd_entry_type_id::RATIONAL)
260  ifd_entry_reader<T,rational<uint16> >::read(r,stream);
261  else if(this->_tid == ifd_entry_type_id::SBYTE)
262  ifd_entry_reader<T,int8>::read(r,stream);
263  else if(this->_tid == ifd_entry_type_id::SSHORT)
264  ifd_entry_reader<T,int16>::read(r,stream);
265  else if(this->_tid == ifd_entry_type_id::SLONG)
266  ifd_entry_reader<T,int32>::read(r,stream);
267  else if(this->_tid == ifd_entry_type_id::SRATIONAL)
268  ifd_entry_reader<T,rational<int32> >::read(r,stream);
269  else if(this->_tid == ifd_entry_type_id::FLOAT)
270  ifd_entry_reader<T,float32>::read(r,stream);
271  else if(this->_tid == ifd_entry_type_id::DOUBLE)
272  ifd_entry_reader<T,float64>::read(r,stream);
273  else
274  //reset stream position
275  throw type_error(EXCEPTION_RECORD,"IFD entry ["+this->name()+
276  "] contains unknown data!");
277  }
