66 template<
67 typename T
68 >
69 class parser
70 {
71 private:
72 using conversion_t = conversion_trait<T>;
73 using read_type = typename conversion_t::read_type;
74 using type_info_t = pni::type_info<T>;
75 public:
76 using result_type = T;
87 result_type operator()(const pni::string &data) const
88 {
89 using namespace pni;
90
91 result_type value;
92
93 if((!type_info_t::is_signed) && (data[0] == '-'))
94 {
95 std::stringstream ss;
96 ss<<"Cannot store a signed value ["<<data<<"] an instance of "
97 <<type_id(value);
98 throw parser_error(PNINEXUS_EXCEPTION_RECORD,ss.str());
99 }
100
101 try
102 {
103 value = conversion_t::convert(
104 boost::lexical_cast<read_type>(data));
105
106 }
107 catch(const boost::bad_lexical_cast &)
108 {
109 std::stringstream ss;
110 ss<<"Could not convert ["<<data<<"] to a value of type ";
111 ss<<type_id(value);
112 throw parser_error(PNINEXUS_EXCEPTION_RECORD,ss.str());
113 }
114 catch(const pni::range_error &error)
115 {
116 std::stringstream ss;
117 ss<<"A range error occured with: "<<error.description()<<std::endl;
118 ss<<"Could not convert input ["<<data<<"] to a value fo type ";
119 ss<<type_id(result_type());
120 throw parser_error(PNINEXUS_EXCEPTION_RECORD,ss.str());
121 }
122 catch(...)
123 {
124 std::stringstream ss;
125 ss<<"Unknown error when trying to convert ["<<data<<"] to ";
126 ss<<type_id(result_type());
127 throw parser_error(PNINEXUS_EXCEPTION_RECORD,ss.str());
128 }
129
130 return value;
131 }
132 };
