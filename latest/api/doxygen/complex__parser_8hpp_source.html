37template<typename BaseT>
38class parser<std::complex<BaseT>>
39{
40 private:
41 using base_type = BaseT;
42 parser<base_type> _base_parser;
43 public:
44 using result_type = std::complex<BaseT>;
45
46 result_type operator()(const pni::string &input) const
47 {
48 using namespace pni;
49
50 boost::regex cmplx_regex(default_complex_regexp);
51
52 result_type value;
53 boost::smatch results;
54 if(boost::regex_match(input,results,cmplx_regex))
55 {
56 base_type real_part = 0.0,
57 imag_part = 0.0,
58 image_sign = 1.0;
59 string part = results.str("REALPART");
60 if(!part.empty()) real_part = _base_parser(part);
61
62 part = results.str("IMAGSIGN");
63 if(!part.empty())
64 {
65 if(part[0]=='-') image_sign = -1.0;
66 }
67
68 part = results.str("IMAGPART");
69 if(!part.empty()) imag_part = _base_parser(part);
70
71 return result_type(real_part,image_sign*imag_part);
72
73 }
74 else
75 {
76 std::stringstream ss;
77 ss<<"Cannot convert ["<<input<<"] to "<<type_id(value)<<"!";
78 throw parser_error(PNINEXUS_EXCEPTION_RECORD,ss.str());
79 }
80 }
81};
82}
