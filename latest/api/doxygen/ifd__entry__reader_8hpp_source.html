61 template<typename ReadT,typename EntryDataT> void ifd_entry_reader<ReadT,EntryDataT>::
62 read(std::vector<ReadT> &r,std::ifstream &stream)
63 {
64
65 EntryDataT buffer;
66
67 //check size of entire entry data
68 if(sizeof(EntryDataT)*r.size()>4){
69 //if the data does not fit into 4 Byte we interpret data as an
70 //offset and move the stream pointer to this new position
71 pni::int32 offset;
72 stream.read((char *)(&offset),4);
73 stream.seekg(offset,std::ios::beg);
74 }
75
76 //read the data
77 for(ReadT &value: r)
78 {
79 stream.read((char*)(&buffer),sizeof(EntryDataT));
80 value = (ReadT)(buffer);
81 }
82
83 }
