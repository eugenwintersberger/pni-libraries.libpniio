61 template<typename IndexIteratorT,
62 typename ShapeIteratorT>
63 static size_t offset(IndexIteratorT &&index_start,
64 IndexIteratorT &&index_stop,
65 ShapeIteratorT &&shape_start)
66 {
67 //initialize the offset and the stride variable
68 size_t offset = *index_start++,stride=1;
69
70 //loop over all indices - remember that the iterators are
71 //reverse iterators
72 while(index_start!=index_stop)
73 {
74 //compute the actuall stride
75 stride *= *shape_start++;
76 //compute the offset contribution
77 offset += stride*(*index_start++);
78 }
79
80 return offset;
81 }

103 template<
104 typename SelectionIteratorT,
105 typename ShapeIteratorT,
106 typename IndexContainerIteratorT

108 static size_t offset(SelectionIteratorT &&sel_shape_start,
109 SelectionIteratorT &&sel_shape_end,

112 ShapeIteratorT &&shape_start,
113 IndexContainerIteratorT &&sel_index)

117 if(*sel_shape_start++ != 1) index += (*sel_index++)*(*sel_stride);

158 template<
159 typename IndexIteratorT,
160 typename ShapeIteratorT
161 >
162 static void index(ShapeIteratorT &&shape_start,
163 ShapeIteratorT &&shape_stop,
164 IndexIteratorT &&index_start,
165 size_t offset)
166 {
167 size_t t;
168 size_t stride = std::accumulate(++shape_start,shape_stop,1,
169 std::multiplies<size_t>());
170 t = offset%stride;
171 *(index_start++) = (offset-t)/stride;
172 offset = t;
173 while(shape_start != shape_stop)
174 {
175 //increment here the shape_start iterator - we already start
176 //with start+1 with the stride computation
177 stride /= *shape_start++;
178 t = offset%stride;
179 *(index_start++) = (offset-t)/stride;
180 offset = t;
181 }
182 }
