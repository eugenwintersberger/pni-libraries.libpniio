317 template<typename ContainerT>
318 void tiff_reader::_read_data(size_t i,size_t c,ContainerT &data)
319 {
320 //obtain the proper IFD
321 tiff::ifd &ifd = this->_ifds.at(i);
322 std::ifstream &stream = this->_get_stream();
323
324 //assume here that the image is stored using strips
325 tiff::strip_reader reader(tiff::strip_reader::create(stream,ifd,this->info(i)));
326 //std::cout<<reader<<std::endl;
327
328 reader.read(c,stream,data);
329
330 }
