36#define PNINEXUS_CHECK_ARITHMETIC_SINGLE(type)\
37 static_assert(std::is_arithmetic<typename type::value_type>::value || \
38 is_complex_type<typename type::value_type>::value,\
39 "Arithemtic operations can only be used with arithmetic types!")
40
41#define PNINEXUS_CHECK_ARITHMETIC_DOUBLE(type1,type2)\
42 static_assert(\
43 (std::is_arithmetic<typename type1::value_type>::value || \
44 is_complex_type<typename type1::value_type>::value) && \
45 (std::is_arithmetic<typename type2::value_type>::value || \
46 is_complex_type<typename type2::value_type>::value), \
47 "Arithemtic operations can only be used with arithmetic types!")
