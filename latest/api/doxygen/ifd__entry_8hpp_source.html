207 //==============implementation of public template methods===================
208 template<typename ElementT> std::vector<ElementT> ifd_entry::value(std::ifstream &stream)
209 {
210 using namespace pni;
211
212 //create a vector of appropriate length
213 std::vector<ElementT> result(this->size());
214 //save the original stream position
215 std::streampos orig_stream_pos = stream.tellg();
216
217 //set the stream to the begining of the data section
218 stream.seekg(this->_data,std::ios::beg);
219
220 //here comes the tricky part. Though the user defines the type ElementT he
221 //wants to have the IFD entry not all entry types can be converted to ElementT
222 //without loss of information. Thus wee need a selection function that
223 //picks for each ElementT all acceptable types and throws an exception if the
224 //entry is of an incompatible type.
225 //
226 //This operation is carried out by the next function which cann choose
227 //the proper version by argument type deduction.
228 try{
229 this->_read_entry_data(result,stream);
230 }
231 catch(type_error &e)
232 {
233 e.append(PNINEXUS_EXCEPTION_RECORD);
234 //reset the stream to its original position
235 stream.seekg(orig_stream_pos,std::ios::beg);
236 throw e;
237 }
238
239
240 //reset stream to its original position
241 stream.seekg(orig_stream_pos,std::ios::beg);
242
243 return result;
244 }

247 template<typename ElementT> void ifd_entry::
248 _read_entry_data(std::vector<ElementT> &r,std::ifstream &stream)
249 {
250 using namespace pni;
251
252 if(this->_tid == ifd_entry_type_id::Byte)
253 ifd_entry_reader<ElementT,uint8>::read(r,stream);
254 else if(this->_tid == ifd_entry_type_id::Short)
255 ifd_entry_reader<ElementT,uint16>::read(r,stream);
256 else if(this->_tid == ifd_entry_type_id::Long)
257 ifd_entry_reader<ElementT,uint32>::read(r,stream);
258 else if(this->_tid == ifd_entry_type_id::Rational)
259 ifd_entry_reader<ElementT,rational<uint16> >::read(r,stream);
260 else if(this->_tid == ifd_entry_type_id::SByte)
261 ifd_entry_reader<ElementT,int8>::read(r,stream);
262 else if(this->_tid == ifd_entry_type_id::SShort)
263 ifd_entry_reader<ElementT,int16>::read(r,stream);
264 else if(this->_tid == ifd_entry_type_id::SLong)
265 ifd_entry_reader<ElementT,int32>::read(r,stream);
266 else if(this->_tid == ifd_entry_type_id::SRational)
267 ifd_entry_reader<ElementT,rational<int32> >::read(r,stream);
268 else if(this->_tid == ifd_entry_type_id::Float)
269 ifd_entry_reader<ElementT,float32>::read(r,stream);
270 else if(this->_tid == ifd_entry_type_id::Double)
271 ifd_entry_reader<ElementT,float64>::read(r,stream);
272 else
273 //reset stream position
274 throw type_error(PNINEXUS_EXCEPTION_RECORD,"IFD entry ["+this->name()+
275 "] contains unknown data!");
276 }
