174 template<typename ContainerT,
175 typename = typename std::enable_if<
176 std::is_same<typename ContainerT::value_type,pni::slice>::value
177 >::type
178 >
179 static array_selection create(const ContainerT &s)
180 {
181 typedef container_utils<index_type> cutils_type;
182 auto shape = cutils_type::create(s.size());
183 auto offset = cutils_type::create(s.size());
184 auto stride = cutils_type::create(s.size());
185
186 size_t index = 0;
187 for(auto sl: s)
188 {
189 offset[index] = sl.first();
190 stride[index] = sl.stride();
191 shape[index] = pni::size(sl);
192 index++;
193 }
194
195 return array_selection(std::move(shape),
196 std::move(offset),
197 std::move(stride));
198 }

410 template<
411 typename ContainerIndexT,
412 typename OriginalIndexT
413 >
414 void array_selection::index(const ContainerIndexT &sindex,OriginalIndexT &oindex) const
415 {
416#ifdef DEBUG
417//check here if the size (rank) of the original shape of the
418 //selection matches that of the original index.
419 check_equal_size(_oshape,oindex,PNINEXUS_EXCEPTION_RECORD);
420 //check if the size of the selection index container matches the
421 //effective rank of the selection
422 check_equal_size(shape<ContainerIndexT>(),sindex,PNINEXUS_EXCEPTION_RECORD);
423#endif
424
425 //now we have to add index*stride from the selection index too
426 //the appropriate locations
427 auto os_iter = _oshape.begin(); //iter. over original shape
428 auto st_iter = _stride.begin(); //iter. over selection strides
429 auto si_iter = sindex.begin(); //iter. over selection index
430 auto of_iter = _offset.begin(); //iter. over the offset
431
432 //loop over output index
433 for(auto &oi: oindex)
434 {
435 size_t index = *of_iter++;
436 //oi = *of_iter++;
437 if(*os_iter++ != 1) index += (*st_iter)*(*si_iter++);
438 ++st_iter; //need to increment this guy in any case
439 oi = index;
440 }
441 }

556 template<typename MapT>
557 bool is_contiguous(const MapT &map,const array_selection &s)
558 {
559
560 size_t orig_first_offset = start_offset(map,s);
561 size_t orig_last_offset = last_offset(map,s);
562
563 //check if the size of the selection matches the offset-difference in
564 //the original array - in this case we have a contiguous selection.
565 if( (orig_last_offset - orig_first_offset + 1) == s.size())
566 return true;
567
568 return false;
569 }
