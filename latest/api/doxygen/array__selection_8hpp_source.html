174  {
175  typedef container_utils<index_type> cutils_type;

177  auto offset = cutils_type::create(s.size());
178  auto stride = cutils_type::create(s.size());

408  void array_selection::index(const ITYPE &sindex,OITYPE &oindex) const
409  {
410 #ifdef DEBUG
411 //check here if the size (rank) of the original shape of the
412  //selection matches that of the original index.
413  check_equal_size(_oshape,oindex,EXCEPTION_RECORD);
414  //check if the size of the selection index container matches the
415  //effective rank of the selection
416  check_equal_size(shape<ITYPE>(),sindex,EXCEPTION_RECORD);
417 #endif
418 
419  //now we have to add index*stride from the selection index too
420  //the appropriate locations
421  auto os_iter = _oshape.begin(); //iter. over original shape
422  auto st_iter = _stride.begin(); //iter. over selection strides
423  auto si_iter = sindex.begin(); //iter. over selection index
424  auto of_iter = _offset.begin(); //iter. over the offset
425 
426  //loop over output index
427  for(auto &oi: oindex)
428  {
429  size_t index = *of_iter++;
430  //oi = *of_iter++;
431  if(*os_iter++ != 1) index += (*st_iter)*(*si_iter++);
432  ++st_iter; //need to increment this guy in any case
433  oi = index;
434  }
435  }

552  {
553 
554  size_t orig_first_offset = start_offset(map,s);
555  size_t orig_last_offset = last_offset(map,s);
556 
557  //check if the size of the selection matches the offset-difference in
