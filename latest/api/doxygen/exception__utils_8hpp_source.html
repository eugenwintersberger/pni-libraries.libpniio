165  template<
166  typename IndexT,
167  typename ShapeT
168  >
169  bool check_indexes(const IndexT &index,const ShapeT &shape)
170  {
171  if(!check_equal_size(index,shape)) return false;
172 
173  auto iiter = index.begin();
174  auto siter = shape.begin();
175 
176  for(;iiter!=index.end();++iiter,++siter)
177  if(*iiter>=*siter) return false;
178 
179  return true;
180  }

296  bool check_equal_shape(const ContainerAT &a,const ContainerBT &b)
297  {
298  //check if the sizes match
299  if(!check_equal_size(a,b)) return false;
300  if(!check_equal_rank(a,b)) return false;
301 
302  auto sa = a.template shape<std::vector<size_t> >();
303  auto sb = b.template shape<std::vector<size_t> >();
304 
305  if(!std::equal(sa.begin(),sa.end(),sb.begin()))
306  return false;
307 
308  return true;
