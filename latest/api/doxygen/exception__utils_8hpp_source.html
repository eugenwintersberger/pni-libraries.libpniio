165 template<
166 typename IndexT,
167 typename ShapeT
168 >
169 bool check_indexes(const IndexT &index,const ShapeT &shape)
170 {
171 if(!check_equal_size(index,shape)) return false;
172
173 auto iiter = index.begin();
174 auto siter = shape.begin();
175
176 for(;iiter!=index.end();++iiter,++siter)
177 if(*iiter>=*siter) return false;
178
179 return true;
180 }

201 template<
202 typename IndexT,
203 typename ShapeT
204 >
205 void check_indexes(const IndexT &index,const ShapeT &shape,
206 const exception_record &record)
207 {
208 //check size - if it does not match throw an exception
209 if(!check_equal_size(index,shape))
210 {
211 std::stringstream ss;
212 ss<<"Rank of index vector ("<<index.size()<<") does not match ";
213 ss<<"the rank of the shape vector ("<<shape.size()<<")!";
214 throw shape_mismatch_error(record,ss.str());
215 }
216
217 //if sizes of the vectors match check the individual ranges.
218 if(!check_indexes(index,shape))
219 {
220 std::stringstream ss;
221 ss<<"Indexes ";
222 print_vector(ss,index);
223 ss<<" do not match shape ";
224 print_vector(ss,shape);
225 ss<<std::endl;
226
227 throw index_error(record,ss.str());
228 }

292 template<
293 typename ContainerAT,
294 typename ContainerBT
295 >
296 bool check_equal_shape(const ContainerAT &a,const ContainerBT &b)
297 {
298 //check if the sizes match
299 if(!check_equal_size(a,b)) return false;
300 if(!check_equal_rank(a,b)) return false;
301
302 auto sa = a.template shape<std::vector<size_t> >();
303 auto sb = b.template shape<std::vector<size_t> >();
304
305 if(!std::equal(sa.begin(),sa.end(),sb.begin()))
306 return false;
307
308 return true;
309 }
