190  //-------------------------------------------------------------------------
191  template<typename BufferT,typename ContainerT>
192  void strip_reader::_read_interlace(size_t channel,
193  std::ifstream &stream,ContainerT &data) const
194  {
195  typedef typename ContainerT::value_type value_type;
196  //compute the size of a pixel in bytes
197  size_t pixel_size = std::accumulate(_bits_per_channel.begin(),
198  _bits_per_channel.end(),0)/8;
199 
200  //compute the offset from the begining of a pixel to the requested
201  //sample. The offset is given in bytes
202  size_t sample_offset = std::accumulate(_bits_per_channel.begin(),
203  _bits_per_channel.begin()+channel,
204  0)/8;
205 
206  //size of the sample type
207  size_t sample_size = sizeof(BufferT);
208 
209  BufferT sample_buffer;
210 
211  typename ContainerT::iterator piter = data.begin(); //pixel iterator
212  while(piter != data.end())
213  {
214  //loop over all strips
215  for(size_t strip=0;strip<_offsets.size();strip++)
216  {
217  size_t npixels = _byte_cnts[strip]/pixel_size;
218 
219  //set the stream to the offset of the actual strip
220  stream.seekg(_offsets[strip]+sample_offset,std::ios::beg);
221 
222  //loop over all pixels in the strip
223  for(size_t i=0;i<npixels; ++i)
224  {
225  stream.read(reinterpret_cast<char*>(&sample_buffer),sample_size);
226  stream.seekg(pixel_size-sample_size,std::ios::cur);
227  *piter++ = value_type(sample_buffer);
228  }
229  }
230  }
231  }
