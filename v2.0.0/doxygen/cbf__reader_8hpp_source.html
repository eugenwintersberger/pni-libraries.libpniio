178 template<typename CTYPE> CTYPE cbf_reader::image(size_t i,size_t c)
179 {
180  using namespace pni;
181  image_info info = _image_info[i];
182  CTYPE data;
183  try
184  {
185  data=CTYPE(info.npixels());
186  }
187  catch(...)
188  {
189  throw memory_allocation_error(EXCEPTION_RECORD,
190  "Allocation of container for image data failed!");
191  }
192 
193  try
194  {
195  image(data,i,c);
196  }
197  catch(file_error &error)
198  {
199  //propagate exception
200  error.append(EXCEPTION_RECORD);
201  throw error;
202  }
203 
204  return data;

208 template<typename CTYPE>
209 void cbf_reader::image(CTYPE &data,size_t i,size_t c)
210 {
211  using namespace pni;
212  //load image information and throw exception if image and container size
213  //to not match
214  image_info inf = _image_info[i];
215  if(data.size()!= inf.npixels())
216  {
217  std::stringstream ss;
218  ss<<"Container size ("<<data.size()<<") does not match image ";
219  ss<<"size ("<<inf.npixels()<<")!";
220  throw size_mismatch_error(EXCEPTION_RECORD,ss.str());
221  }
222 
223  //load the channel information
224  image_channel_info channel = inf.get_channel(c);
225 
226  if(_detector_vendor == cbf::vendor_id::DECTRIS)
227  {
228  if(channel.type_id() == type_id_t::INT16)
229  //read 16Bit signed data
230  cbf::dectris_reader::read_data_byte_offset<int16>(
231  _get_stream(),inf,data);
232  if(channel.type_id() == type_id_t::INT32)
233  //read 32Bit signed data
234  cbf::dectris_reader::read_data_byte_offset<int32>(
235  _get_stream(),inf,data);
236  else
237  {
238  file_error error(EXCEPTION_RECORD,
239  "No data reader for this data type!");
240  throw error;
241  }
242 
243  }
244  else
245  throw file_error(EXCEPTION_RECORD,"Unknown detector vendor!");
246 
247 }
248 //end of namespace
