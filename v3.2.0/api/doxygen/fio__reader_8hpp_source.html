257 //-------------------------------------------------------------------------
258 template<typename ContainerT>
259 void fio_reader::column(const std::string &n,ContainerT &c) const
260 {
261 using namespace pni;
262 using value_type = typename ContainerT::value_type;
263
264 try
265 {
266 const column_type &column = _columns.at(n);
267
268 size_t index=0;
269 for(auto value: column)
270 c[index++] = boost::lexical_cast<value_type>(value);
271 }
272 catch(key_error &error)
273 {
274 //append a new issuer to the exception
275 error.append(PNINEXUS_EXCEPTION_RECORD);
276 throw error;
277 }
278 }
