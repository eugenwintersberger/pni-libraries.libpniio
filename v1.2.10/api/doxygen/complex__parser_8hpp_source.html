48  {

50 
51  boost::regex cmplx_regex(default_complex_regexp);
52 
53  result_type value;
54  boost::smatch results;
55  if(boost::regex_match(input,results,cmplx_regex))
56  {
57  base_type real_part = 0.0,
58  imag_part = 0.0,
59  image_sign = 1.0;
60  string part = results.str("REALPART");
61  if(!part.empty()) real_part = _base_parser(part);
62 
63  part = results.str("IMAGSIGN");
64  if(!part.empty())
65  {
66  if(part[0]=='-') image_sign = -1.0;
67  }
68 
69  part = results.str("IMAGPART");
70  if(!part.empty()) imag_part = _base_parser(part);
71 
72  return result_type(real_part,image_sign*imag_part);
73 
74  }
75  else
76  {
77  std::stringstream ss;
78  ss<<"Cannot convert ["<<input<<"] to "<<type_id(value)<<"!";
79  throw parser_error(EXCEPTION_RECORD,ss.str());
80  }
81  }
82 };
