89  result_type operator()(const pni::core::string &input) const
90  {
91  using namespace pni::core;
92  auto first = input.begin();
93  auto last = input.end();
94  //need to find the start and stop
95  if(_config.start_symbol())
96  {
97  first = std::find(first,last,_config.start_symbol());
98  if(first==last)
99  {
100  std::stringstream ss;
101  ss<<"Input: "<<input<<" - has no start symbol!";
102  throw parser_error(EXCEPTION_RECORD,ss.str());
103  }
104  std::advance(first,1);
105  }
106 
107  if(_config.stop_symbol())
108  {
109  last = std::find(first,last,_config.stop_symbol());
110  if(last==input.end())
111  {
112  std::stringstream ss;
113  ss<<"Input: "<<input<<" - has no stop symbol!";
114  throw parser_error(EXCEPTION_RECORD,ss.str());
115  }
116  }
117 
118  string buffer(first,last);
119  std::vector<string> elements;
120  string sep(1,_config.separator());
121  boost::split(elements,buffer, boost::is_any_of(sep),
122  boost::token_compress_on);
123  std::transform(elements.begin(),elements.end(),elements.begin(),
124  [](string &value) { boost::trim(value); return value; });
125  auto new_end = std::remove_if(elements.begin(),elements.end(),
126  [](const string &value) { return value.empty(); });
127 
128  result_type result;
129  std::transform(elements.begin(),new_end,
130  std::back_inserter(result),
131  _value_parser);
132 
133  return result;
134  }
