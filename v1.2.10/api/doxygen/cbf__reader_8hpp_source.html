179 template<typename CTYPE> CTYPE cbf_reader::image(size_t i,size_t c)
180 {
181  using namespace pni::core;
182  image_info info = _image_info[i];
183  CTYPE data;
184  try
185  {
186  data=CTYPE(info.npixels());
187  }
188  catch(...)
189  {
190  throw memory_allocation_error(EXCEPTION_RECORD,
191  "Allocation of container for image data failed!");
192  }
193 
194  try
195  {
196  image(data,i,c);
197  }
198  catch(file_error &error)
199  {
200  //propagate exception
201  error.append(EXCEPTION_RECORD);
202  throw error;
203  }
204 
205  return data;
206 }

209 template<typename CTYPE>
210 void cbf_reader::image(CTYPE &data,size_t i,size_t c)
211 {
212  using namespace pni::core;
213  //load image information and throw exception if image and container size
214  //to not match
215  image_info inf = _image_info[i];
216  if(data.size()!= inf.npixels())
217  {
218  std::stringstream ss;
219  ss<<"Container size ("<<data.size()<<") does not match image ";
220  ss<<"size ("<<inf.npixels()<<")!";
221  throw size_mismatch_error(EXCEPTION_RECORD,ss.str());
222  }
223 
224  //load the channel information
225  image_channel_info channel = inf.get_channel(c);
226 
227  if(_detector_vendor == cbf::vendor_id::DECTRIS)
228  {
229  if(channel.type_id() == type_id_t::INT16)
230  //read 16Bit signed data
231  cbf::dectris_reader::read_data_byte_offset<int16>(
232  _get_stream(),inf,data);
233  if(channel.type_id() == type_id_t::INT32)
234  //read 32Bit signed data
235  cbf::dectris_reader::read_data_byte_offset<int32>(
236  _get_stream(),inf,data);
237  else
238  {
239  file_error error(EXCEPTION_RECORD,
240  "No data reader for this data type!");
241  throw error;
242  }
243 
244  }
245  else
246  throw file_error(EXCEPTION_RECORD,"Unknown detector vendor!");
247 
248 }
249 //end of namespace
250 }
251 }
